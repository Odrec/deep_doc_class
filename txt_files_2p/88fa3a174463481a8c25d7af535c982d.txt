togram

load('spiking.mat')

figure % spike trains rasterplot
for j = 1:10
    for i = 1:length(neuron(j).times)
        line([neuron(j).times(i) neuron(j).times(i)], [j-1 j]);
    end
end
title('Rasterplot for the first 10 trials');
xlabel('time, 0 beginning of the movement');
ylabel('Number of trial');

figure % Histogram
bins = linspace(-1,1,21); % 21 values to get 20 bins
n = zeros(21, length(neuron)); % output
for i = 1:length(neuron)
        n(:, i) = histc(neuron(i).times, bins);
end
average = mean(n,2);

% bins must be of length 21 to specify only 20 buckets because
% histc uses the intervals between the values in bins -> the last bucket
% will always be 0. We have to plot only the first 20 values
bar(linspace(-1,1,20), average(1:20), 'BarWidth', 1);
xlim ([-1 1]);
title('PSTH');
xlabel('time, 0 beginning of the movement');
ylabel('spikes');
