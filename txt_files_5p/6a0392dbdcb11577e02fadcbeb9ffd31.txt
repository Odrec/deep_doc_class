%c changing fixations to fit to the cropped images
x1_minus = fixations.x(fixations.subject==1) - 50;
y1_minus = fixations.y(fixations.subject==1) - 50;
x2_minus = fixations.x(fixations.subject==2) - 50;
y2_minus = fixations.y(fixations.subject==2) - 50;
x3_minus = fixations.x(fixations.subject==3) - 50;
y3_minus = fixations.y(fixations.subject==3) - 50;
x4_minus = fixations.x(fixations.subject==4) - 50;
y4_minus = fixations.y(fixations.subject==4) - 50;
% adding together all participants fixation points
fix_x_allsub = [x1_minus x2_minus x3_minus x4_minus];
fix_y_allsub = [y1_minus y2_minus y3_minus y4_minus];

% for each fixation of each subject, find the value on the contrast map and
% put into a matrix
fix_rg = [];
fix_by = [];
fix_lum = [];
for i=1:length(fix_x_allsub),
    if fix_x_allsub(i)>0 & fix_x_allsub(i)<=669 & fix_y_allsub(i)>0 & fix_y_allsub(i)<=925
        % only use fixation points that are in the cropped out version
        fix_rg = [fix_rg rg_crop(round(fix_x_allsub(i)),round(fix_y_allsub(i)))];

4

        fix_by = [fix_by by_crop(round(fix_x_allsub(i)),round(fix_y_allsub(i)))];
        fix_lum = [fix_lum lum_crop(round(fix_x_allsub(i)),round(fix_y_allsub(i)))];
    end
end

% plotting the fixations
