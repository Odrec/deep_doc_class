% Orientation map describes the preferred angle for every neuron with
% respect to the relative firing rates, whereas the activation map
% describes the activity of a group of neurons at one common angle.
% They both measure spike activation rates in neurons, dependent on angles,
% and coded in different colours.

act_map = importdata('act.mat')

for i=1:50;
    for j=1:50;
        max = 0;
        bestk = 1;
        for k=1:8

                max=act_map(i,j,k);
                bestk=k+1
            end
        end

    end
end
