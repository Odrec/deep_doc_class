Making the ROC-curve

roc_y = [];
roc_x = [];

% for-loop to let the threshold slide over courve
% threshold starts at no-stimulus minimum since here TP = ~0 and ends at
% with-stimulus maximum since here FP = ~0
for threshold = min(ns(:)):max(ws(:)),
   tpr = sum(ws(ws > threshold))/sum(ws);
   roc_y(end+1) = tpr;

   fpr = sum(ns(ns > threshold))/sum(ns);
   roc_x(end+1) = fpr;
end

figure
plot(roc_x,roc_y,'x');
hold on
refline([1,0]);
hold off
xlabel('FalsePositiveRate');
ylabel('TruePositiveRate');
