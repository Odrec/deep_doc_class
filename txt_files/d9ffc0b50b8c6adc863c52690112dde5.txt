Compute orientation tuning curve

% compute orientations in degrees for x-axis
x = linspace(0,157.5,8);

%count spikes for each orientation
spike_count = sum(d,3);

%take mean count over the trials
average_spikes = mean(spike_count,2);

Visualize data

x = x';

figure;
plot(x,average_spikes,'b.')
xlabel('Orientation in Degrees')
ylabel('Mean Spike Count')
title('Tuning Curve')
